const FadeIn = ({ children, delay = 0, direction = "up", className = "", fullWidth = false }) => {

  const [isVisible, setIsVisible] = useState(false);

  const domRef = useRef();


  useEffect(() => {

    const observer = new IntersectionObserver(entries => {

      if (entries[0].isIntersecting) {

        setIsVisible(true);

        if (domRef.current) observer.unobserve(domRef.current); // 确保只动画一次

      }

    }, { 

      threshold: 0.1, 

      rootMargin: '0px 0px -50px 0px' 

    });


    if (domRef.current) observer.observe(domRef.current);

    return () => {

      if (domRef.current) observer.disconnect();

    };

  }, []);


  const transitionClass = `transition-all duration-[1200ms] ease-[cubic-bezier(0.16,1,0.3,1)]`;

  const delayStyle = { transitionDelay: `${delay}ms` };

  

  let startTransform = "";

  if (direction === "up") startTransform = "translate-y-12";

  if (direction === "down") startTransform = "-translate-y-12";

  if (direction === "left") startTransform = "translate-x-12";

  if (direction === "right") startTransform = "-translate-x-12";


  const activeClass = isVisible ? "opacity-100 translate-y-0 translate-x-0" : `opacity-0 ${startTransform}`;


  return (

    <div ref={domRef} className={`${transitionClass} ${activeClass} ${fullWidth ? 'w-full' : ''} ${className}`} style={delayStyle}>

      {children}

    </div>

  );

};

const CustomCursor = () => {

  const cursorRef = useRef(null);

  const [isHovering, setIsHovering] = useState(false);

  const [isTouchDevice, setIsTouchDevice] = useState(false);


  useEffect(() => {

    // 检测是否为触屏设备，移动端不启用自定义光标以防止交互残留

    if ('ontouchstart' in window || navigator.maxTouchPoints > 0) {

      setIsTouchDevice(true);

      return;

    }


    const cursor = cursorRef.current;

    if (!cursor) return;


    const onMouseMove = (e) => {

      // 将 24px 宽高的圆点中心精确对准鼠标指针 (偏移 12px)

      cursor.style.transform = `translate3d(${e.clientX - 12}px, ${e.clientY - 12}px, 0)`;

    };


    const onMouseOver = (e) => {

      const target = e.target;

      // 检测鼠标是否悬停在可交互元素上，用于触发放大动效

      if (

        target.tagName.toLowerCase() === 'button' ||

        target.tagName.toLowerCase() === 'a' ||

        target.closest('button') ||

        target.closest('a') ||

        window.getComputedStyle(target).cursor === 'pointer'

      ) {

        setIsHovering(true);

      } else {

        setIsHovering(false);

      }

    };


    window.addEventListener('mousemove', onMouseMove);

    window.addEventListener('mouseover', onMouseOver);


    return () => {

      window.removeEventListener('mousemove', onMouseMove);

      window.removeEventListener('mouseover', onMouseOver);

    };

  }, []);


  if (isTouchDevice) return null;


  return (

    <>

      <style dangerouslySetInnerHTML={{__html: `

        /* 仅在支持精准指针的设备上隐藏系统默认光标 */

        @media (pointer: fine) {

          * { cursor: none !important; }

        }

      `}} />

      {/* 光标定位容器，剥离动画以保证零延迟跟随 */}

      <div

        ref={cursorRef}

        className="fixed top-0 left-0 w-6 h-6 pointer-events-none z-[99999] will-change-transform flex items-center justify-center"

        style={{ transform: 'translate3d(-100px, -100px, 0)' }} // 初始状态隐藏在屏幕外

      >

        {/* 光标视觉实体，处理悬停时的缩放与色彩过渡 */}

        <div 

          className={`w-full h-full rounded-full bg-[#C0FF04]/60 backdrop-blur-[2px] shadow-[0_0_10px_rgba(192,255,4,0.3)] transition-all duration-300 ease-[cubic-bezier(0.16,1,0.3,1)] ${

            isHovering ? 'scale-[1.8] bg-[#C0FF04]/30' : 'scale-100'

          }`}

        />

      </div>

    </>

  );

};

// 包含移动端零边距与滑动切换交互优化

const GalleryLightbox = ({ images, currentIndex, onClose, onNavigate }) => {

  const overlayRef = useRef(null);

  

  // 移动端手势滑动状态

  const [touchStart, setTouchStart] = useState(null);

  const [touchEnd, setTouchEnd] = useState(null);

  const minSwipeDistance = 40; // 触发切换的最小滑动距离(px)


  useEffect(() => {

    if (currentIndex === null) return;


    document.body.style.overflow = 'hidden';


    const handleKeyDown = (e) => {

      if (e.key === 'Escape') onClose();

      // 支持键盘左右键无限循环

      if (e.key === 'ArrowRight') onNavigate((currentIndex + 1) % images.length);

      if (e.key === 'ArrowLeft') onNavigate((currentIndex - 1 + images.length) % images.length);

    };


    const el = overlayRef.current;

    let isWheeling = false;

    const handleWheel = (e) => {

      e.preventDefault(); 

      if (isWheeling) return;

      

      if (Math.abs(e.deltaX) > 25) {

        isWheeling = true;

        // 支持鼠标横向滚轮无限循环

        if (e.deltaX > 0) {

          onNavigate((currentIndex + 1) % images.length);

        } else if (e.deltaX < 0) {

          onNavigate((currentIndex - 1 + images.length) % images.length);

        }

        setTimeout(() => { isWheeling = false; }, 400);

      }

    };


    window.addEventListener('keydown', handleKeyDown);

    if (el) el.addEventListener('wheel', handleWheel, { passive: false });


    return () => {

      document.body.style.overflow = '';

      window.removeEventListener('keydown', handleKeyDown);

      if (el) el.removeEventListener('wheel', handleWheel);

    };

  }, [currentIndex, images.length, onClose, onNavigate]);


  // 处理触屏滑动事件

  const onTouchStart = (e) => {

    setTouchEnd(null); // 重置滑动结束位置

    setTouchStart(e.targetTouches[0].clientX);

  };


  const onTouchMove = (e) => {

    setTouchEnd(e.targetTouches[0].clientX);

  };


  const onTouchEnd = () => {

    if (!touchStart || !touchEnd) return;

    const distance = touchStart - touchEnd;

    const isLeftSwipe = distance > minSwipeDistance;

    const isRightSwipe = distance < -minSwipeDistance;


    // 支持移动端手势无限循环

    if (isLeftSwipe) {

      onNavigate((currentIndex + 1) % images.length);

    }

    if (isRightSwipe) {

      onNavigate((currentIndex - 1 + images.length) % images.length);

    }

  };


  if (currentIndex === null || images.length === 0) return null;


  return (

    <div 

      ref={overlayRef}

      className="fixed inset-0 z-[100] flex items-center justify-center bg-black/95 backdrop-blur-xl transition-opacity duration-300 overscroll-none touch-none"

      onTouchStart={onTouchStart}

      onTouchMove={onTouchMove}

      onTouchEnd={onTouchEnd}

    >

      <button 

        onClick={onClose}

        className="absolute top-4 right-4 md:top-5 md:right-5 p-2 md:p-2.5 text-white/50 hover:text-white transition-colors z-50 rounded-full hover:bg-white/10"

      >

        <XIcon className="w-7 h-7 md:w-8 md:h-8" />

      </button>


      {/* 优化：取消首尾限制，只要有多张图就显示左右侧箭头，并应用取模循环算法 */}

      {images.length > 1 && (

        <button 

          onClick={(e) => { e.stopPropagation(); onNavigate((currentIndex - 1 + images.length) % images.length); }}

          className="hidden md:block absolute left-2 md:left-4 p-2 md:p-3 text-white/50 hover:text-white transition-colors z-50 rounded-full hover:bg-white/10"

        >

          <ChevronLeftIcon className="w-8 h-8 md:w-10 md:h-10" />

        </button>

      )}


      {/* 核心视觉优化：移动端(px-0)完全取消安全内边距让图片贴合屏幕边缘；PC端(md:px-20)保留原有留白 */}

      <div 

        className="relative w-full h-full flex items-center justify-center select-none px-0 py-16 md:px-20 md:py-20"

        onClick={onClose}

      >

        <img 

          src={images[currentIndex]} 

          alt={`Gallery View ${currentIndex + 1}`} 

          className="max-w-full max-h-full object-contain md:rounded-md shadow-2xl md:shadow-none pointer-events-none transition-transform duration-500 ease-out"

          key={currentIndex}

        />

      </div>


      {images.length > 1 && (

        <button 

          onClick={(e) => { e.stopPropagation(); onNavigate((currentIndex + 1) % images.length); }}

          className="hidden md:block absolute right-2 md:right-4 p-2 md:p-3 text-white/50 hover:text-white transition-colors z-50 rounded-full hover:bg-white/10"

        >

          <ChevronRightIcon className="w-8 h-8 md:w-10 md:h-10" />

        </button>

      )}


      {/* 仅在多图集合模式下呈现当前进度指示器 */}

      {images.length > 1 && (

        <div className="absolute bottom-5 md:bottom-6 left-1/2 -translate-x-1/2 text-white/40 font-mono text-xs md:text-sm tracking-widest pointer-events-none">

          {currentIndex + 1} / {images.length}

        </div>

      )}

    </div>

  );

};

const Photography3DGallery = ({ images, onImageClick }) => {

  const containerRef = useRef(null);

  const itemsRef = useRef([]);

  

  // 修正焦点公式：因为左侧排了15张图，真正的居中索引是 15 (即 30 / 2)

  const initialIndex = Math.floor(images.length / 2);

  const targetProgress = useRef(initialIndex);

  const currentProgress = useRef(initialIndex);

  const isDragging = useRef(false);

  const startX = useRef(0);

  const startProgress = useRef(0);

  const requestRef = useRef(null);


  useEffect(() => {

    return () => {

      document.body.style.overscrollBehaviorX = 'auto'; 

    };

  }, []);


  const handleMouseEnter = () => {

    document.body.style.overscrollBehaviorX = 'none';

  };


  const handleMouseLeave = () => {

    isDragging.current = false;

    if (containerRef.current) {

      containerRef.current.style.cursor = 'grab';

    }

    document.body.style.overscrollBehaviorX = 'auto';

  };


  useEffect(() => {

    itemsRef.current = itemsRef.current.slice(0, images.length);

  }, [images]);


  useEffect(() => {

    const renderFrame = () => {

      if (!isDragging.current) {

        const nearestIndex = Math.round(targetProgress.current);

        if (Math.abs(targetProgress.current - nearestIndex) < 0.5) {

          targetProgress.current += (nearestIndex - targetProgress.current) * 0.05;

        }

      }

      

      currentProgress.current += (targetProgress.current - currentProgress.current) * 0.1;


      itemsRef.current.forEach((item, index) => {

        if (!item) return;

        

        const offset = index - currentProgress.current;

        const absOffset = Math.abs(offset);

        const sign = Math.sign(offset);


        if (absOffset > 7) {

          item.style.display = 'none';

          return;

        } else {

          item.style.display = 'block';

        }


        // 核心视觉优化：重构 3D 转换参数，适配全新的 1:1 正方形卡片比例

        const isMobile = window.innerWidth < 768;

        // 1. 配合 1:1 更宽的卡片，微调横向基础间距以保持叠放美感

        const translateX = sign * (absOffset * (isMobile ? 45 : 85)); 

        // 2. 加深 Z 轴景深，提升立体感

        const translateZ = -absOffset * 90; 

        // 3. 优化旋转角度

        const rotateY = sign * Math.min(absOffset * 48, 55); 

        // 4. 调整层级缩放

        const scale = Math.max(1 - absOffset * 0.05, 0.4);

        const opacity = Math.max(1 - absOffset * 0.15, 0);


        item.style.transform = `translateX(${translateX}px) translateZ(${translateZ}px) rotateY(${rotateY}deg) scale(${scale})`;

        item.style.opacity = opacity;

        item.style.zIndex = 100 - Math.round(absOffset * 10);

        

        // 仅对图片元素应用暗化滤镜，避免污染容器底色和边缘阴影

        const img = item.querySelector('img');

        if (img) {

          const brightness = Math.max(1 - absOffset * 0.1, 0.6);

          img.style.filter = `brightness(${brightness})`;

        }

      });


      requestRef.current = requestAnimationFrame(renderFrame);

    };


    requestRef.current = requestAnimationFrame(renderFrame);

    return () => cancelAnimationFrame(requestRef.current);

  }, [images.length]);


  useEffect(() => {

    const container = containerRef.current;

    if (!container) return;


    const handleWheel = (e) => {

      if (Math.abs(e.deltaX) > Math.abs(e.deltaY)) {

        e.preventDefault(); 

        const maxIndex = images.length - 1;

        targetProgress.current = Math.max(0, Math.min(maxIndex, targetProgress.current + e.deltaX * 0.02));

      }

    };


    container.addEventListener('wheel', handleWheel, { passive: false });

    return () => container.removeEventListener('wheel', handleWheel);

  }, [images.length]);


  const handlePointerDown = (e) => {

    isDragging.current = true;

    startX.current = e.clientX || (e.touches && e.touches[0].clientX);

    startProgress.current = targetProgress.current;

    if (containerRef.current) {

      containerRef.current.style.cursor = 'grabbing';

    }

  };


  const handlePointerMove = (e) => {

    if (!isDragging.current) return;

    if (e.cancelable && e.type === 'touchmove') e.preventDefault(); 


    const clientX = e.clientX || (e.touches && e.touches[0].clientX);

    const deltaX = startX.current - clientX;

    

    const sensitivity = window.innerWidth < 768 ? 0.015 : 0.005; 

    targetProgress.current = Math.max(0, Math.min(images.length - 1, startProgress.current + deltaX * sensitivity));

  };


  const handleClick = (index) => {

    targetProgress.current = index;

    if (onImageClick) {

      onImageClick(index);

    }

  };


  return (

    <div className="relative w-full">

      <div 

        ref={containerRef}

        // 重构：基于 1:1 比例要求重新推断合适的基础高度容器

        className="relative w-full h-[55vw] sm:h-[350px] md:h-[450px] lg:h-[500px] xl:h-[550px] cursor-grab touch-none"

        onMouseEnter={handleMouseEnter}

        onMouseLeave={handleMouseLeave}

        onMouseDown={handlePointerDown}

        onMouseMove={handlePointerMove}

        onMouseUp={handleMouseLeave}

        onTouchStart={handlePointerDown}

        onTouchMove={handlePointerMove}

        onTouchEnd={handleMouseLeave}

      >

        <div 

          className="absolute inset-0 flex items-center justify-center pointer-events-none"

          // 核心修复：移除 preserve-3d 关闭物理碰撞，将透视点移至此处，强制浏览器尊重 zIndex 的安全叠放

          style={{ perspective: '1500px' }}

        >

          {images.map((img, idx) => (

            <div

              key={idx}

              ref={el => itemsRef.current[idx] = el}

              onClick={() => handleClick(idx)}

              // 核心修复1：使用显式的 width 和 height (赋予完全相等的值) 彻底锁死 1:1 正方形比例，根除拉伸 bug

              className="absolute w-[55vw] sm:w-[280px] md:w-[340px] lg:w-[400px] xl:w-[460px] h-[55vw] sm:h-[280px] md:h-[340px] lg:h-[400px] xl:h-[460px] rounded-xl md:rounded-2xl overflow-hidden shadow-[0_20px_60px_rgba(0,0,0,0.15)] bg-transparent cursor-pointer select-none origin-center transition-all duration-[800ms] ease-[cubic-bezier(0.16,1,0.3,1)]"

              style={{ pointerEvents: 'auto', willChange: 'transform, opacity' }}

            >

              <img 

                src={img} 

                alt={`Photography sequence ${idx + 1}`} 

                // 核心修复2：添加 absolute inset-0 配合 object-cover 防止图片在绝对定位盒模型内出现意外缩放

                className="absolute inset-0 block w-full h-full object-cover pointer-events-none" 

                loading="lazy"

                draggable="false"

                style={{ 

                  willChange: 'filter',

                  WebkitTransform: 'translateZ(0)',

                  transform: 'translateZ(0)' 

                }}

              />

            </div>

          ))}

        </div>

      </div>

    </div>

  );

};
